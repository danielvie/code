Example: AppComponent keyboard tracking bootstrap

Before 6fc4780f5d 
import { Component, ViewEncapsulation, OnInit } from "@angular/core";
import { PrimeNGConfig } from 'primeng/api';

export class AppComponent implements OnInit {

  constructor(private primengConfig: PrimeNGConfig) {}

  ngOnInit() {
    this.primengConfig.ripple = true;
  }
}
After 
import { Component, OnDestroy, OnInit, ViewEncapsulation } from "@angular/core";
import { PrimeNGConfig } from 'primeng/api';
import { RefresherService } from './refresher/refresher.service';

export class AppComponent implements OnInit, OnDestroy {

  constructor(
    private primengConfig: PrimeNGConfig,
    private refresherService: RefresherService
  ) {}

  ngOnInit() {
    this.primengConfig.ripple = true;
    this.refresherService.initializeControlKeyTracking();
  }

  ngOnDestroy() {
    this.refresherService.destroyControlKeyTracking();
  }
}

The root component now bootstraps a shared keyboard tracker and tears it down on destroy.

Example: Replacing @HostListener with scoped listener

Before 6fc4780f5d 
@HostListener('document:keydown', ['$event'])
handleKeyboardEvent(event: KeyboardEvent) {
  if ((event.ctrlKey || event.metaKey) && event.key === 'c') {
    event.preventDefault();
    if (this.multiSelectedAttributes.length > 0) {
      const table = this.generateTable(this.multiSelectedAttributes);
      this.copyToClipBoard(table);
    }
  }
}
After 
private removeGlobalKeydownListener?: () => void;

ngOnInit() { this.setupGlobalKeydownListener(); }
ngOnDestroy() { this.removeGlobalKeydownListener?.(); }

private setupGlobalKeydownListener() {
  this._ngZone.runOutsideAngular(() => {
    const listener = (event: KeyboardEvent) => {
      const isCopy = this.isTabActive && event.ctrlKey && event.key === 'c';
      if (!isCopy) return;
      this._ngZone.run(() => this.handleCopyShortcut(event));
    };
    document.addEventListener('keydown', listener);
    this.removeGlobalKeydownListener = () =>
      document.removeEventListener('keydown', listener);
  });
}

The @HostListener is replaced with an explicit listener gated by isTabActive and properly cleaned up on destroy.
